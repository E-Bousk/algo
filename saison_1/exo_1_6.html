Exercice 1.6

Enoncé

« Ecrire un pseudo-code permettant d’échanger les valeurs de deux variables A et B, et ce quel que soit leur contenu préalable. »

PSEUDO-CODE

// ALGORITHME Exo_1_6 JSPROMPT
function Exo_1_6() 
// DEBUT
{
	//VALEUR sVarA, sVarB, sTemp
	var sVarA, sVarB, sTemp;

	// ECRIRE "Entrez la valeur A"
	// LIRE sVarA
	sVarA= prompt("Entrez la valeur A :");

	// ECRIRE "Entrez la valeur B"
	// LIRE sVarB
	sVarB= prompt("Entrez la valeur B :");

	//Inversion des valeurs
	sTemp=sVarA;
	sVarA=sVarB;
	sVarB=sTemp;

	//Ecrire valeur A, B
	alert ("Inversion des valeurs :" + '\n' + "A = " + sVarA + '\n' + "B = " + sVarB);
//Fin
}
