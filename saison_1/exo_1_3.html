Exercice 1.3

Enoncé

« Quelles seront les valeurs des variables A et B après exécution des instructions suivantes ? »

PSEUDO-CODE

// ALGORITHME Exo_1_3 JSPROMPT
function Exo_1_3() 
    // DEBUT
    {
        //VALEUR A, B
        var A, B;

        //attribution des valeurs
        A=49;
        B=A+4;
        A=A+1;
        B=A-4;

        //Ecrire valeur A, B
        alert ("A = " + A + '\n' + "B = " + B);
    //Fin
    }						
// ALGORITHME Exo_1_3 JSFORM
function Exo_1_3_jsform() 
    // DEBUT
    {
        //VALEUR A, B
        var A, B;

        //attribution des valeurs
        A=49;
        B=A+4;
        A=A+1;
        B=A-4;

        //Ecrire valeur A, B
        document.getElementById("sp_resultat_code").innerHTML= ("A = " + A + "

    //Fin
    }

// ALGORITHME Exo_1_3 PHP - Appel AJAX
function Exo_1_3_phpAjax() {
	$.ajax(
		{
			type: "POST", 
			url: "exo_1_3_Ajax.php", 
			async: true, 
			// data: datas, 
			dataType: "json", 
			cache: false, 
		})
	
		.done(function(result) 
		{
			$("#sp_resultat_code").html("A = " + result['A'] + "
 B = "+ result['B']);
		})
	
		.fail(function(err) 
		{
			// Affichage d'erreur
			console.log("fail");
			alert("error : " + err.status);
		});
}
