// ALGORITHME Exo_1_4 PHP - Appel AJAX
function Exo_1_5_phpAjax() {
	$.ajax(
		{
			type: "POST", 
			url: "exo_1_5_Ajax.php", 
			async: true, 
			// data: datas, 
			dataType: "json", 
			cache: false, 
		})
	
		.done(function(result)

			$("#sp_resultat_code").html("A = " + result['A'] + "
 B = "+ result['B']);
		})
	
		.fail(function(err) 
		{
			// Affichage d'erreur
			console.log("fail");
			alert("error : " + err.status);
		});
}
