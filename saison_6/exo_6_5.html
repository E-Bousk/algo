// ALGORITHME Exo_6_5 JSFORM
function Exo_6_5_jsform()
//Début
{
    var Tab, i;
    i=1;
    Tab = [6];
    Tab[0]=1;

    // Ecrire Tab(0)
    document.getElementById("sp_result").innerHTML=Tab[0] + " , ";

    // Pour i ← 1 à 6
    for (i;i<=6;i++)
    {
        Tab[i]=Tab[i-1]+2;
        // Ecrire Tab(i)
        document.getElementById("sp_result").innerHTML+=Tab[i] + ((i<6) ? " , " :"");
    // i Suivant	
    }
//Fin
}
