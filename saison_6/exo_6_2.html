// ALGORITHME Exo_6_2 JSFORM
function Exo_6_2_jsform()
//Début
{
	// Effacer résultat précédent
	document.getElementById("sp_result").innerHTML="";

	var Tab = [5];

	Tab[0] = "A";
	Tab[1] = "E";
	Tab[2] = "I";
	Tab[3] = "O";
	Tab[4] = "U";
	Tab[5] = "Y";

	//Ecrire "Tableau :" & Tab
	document.getElementById("sp_result").innerHTML="Tableau : " + Tab + "<br>";

//Fin
}
